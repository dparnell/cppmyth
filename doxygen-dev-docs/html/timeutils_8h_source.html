80  inline int64_t GetTimeMs()
81  {
82  #if defined(__APPLE__)
83  int64_t ticks = 0;
84  static mach_timebase_info_data_t timebase;
85  if (timebase.denom == 0) {
86  // Get the timebase if this is the first time we run.
87  // Recommended by Apple's QA1398.
88  (void)mach_timebase_info(&timebase);
89  }
90  // Use timebase to convert absolute time tick units into nanoseconds.
91  ticks = mach_absolute_time() * timebase.numer / timebase.denom;
92  return ticks / 1000000;
93  #elif defined(__WINDOWS__)
94  LARGE_INTEGER tickPerSecond;
95  LARGE_INTEGER tick;
96  if (QueryPerformanceFrequency(&tickPerSecond))
97  {
98  QueryPerformanceCounter(&tick);
99  return (int64_t) (tick.QuadPart / (tickPerSecond.QuadPart / 1000.));
100  }
101  return -1;
102  #else
103  timespec time;
104  clock_gettime(CLOCK_MONOTONIC, &time);
105  return (int64_t)time.tv_sec * 1000 + time.tv_nsec / 1000000;
106  #endif
107  }
