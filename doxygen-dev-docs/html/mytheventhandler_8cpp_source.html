143 bool BasicEventHandler::SubscribeForEvent(unsigned subid, EVENT_t event)
144 {
145  PLATFORM::CLockObject lock(*m_mutex);
146  // Only for registered subscriber
147  subscriptions_t::const_iterator it1 = m_subscriptions.find(subid);
148  if (it1 == m_subscriptions.end())
149  return false;
150  std::vector<unsigned>::const_iterator it2 = m_subscriptionsByEvent[event].begin();
151  while (it2 != m_subscriptionsByEvent[event].end())
152  {
153  if (*it2 == subid)
154  return true;
155  ++it2;
156  }
157  m_subscriptionsByEvent[event].push_back(subid);
158  return true;
159 }

170 void BasicEventHandler::DispatchEvent(const EventMessage& msg)
171 {
172  PLATFORM::CLockObject lock(*m_mutex);
173  std::vector<std::vector<unsigned>::iterator> revoked;
174  std::vector<unsigned>::iterator it1 = m_subscriptionsByEvent[msg.event].begin();
175  while (it1 != m_subscriptionsByEvent[msg.event].end())
176  {
177  subscriptions_t::const_iterator it2 = m_subscriptions.find(*it1);
178  if (it2 != m_subscriptions.end())
179  it2->second->HandleBackendMessage(msg);
180  else
181  revoked.push_back(it1);
182  ++it1;
183  }
184  std::vector<std::vector<unsigned>::iterator>::const_iterator itr;
185  for (itr = revoked.begin(); itr != revoked.end(); ++itr)
186  m_subscriptionsByEvent[msg.event].erase(*itr);
187 }

189 void *BasicEventHandler::Process()
190 {
191  // Try to connect
192  if (m_event->Open())
193  AnnounceStatus(EVENTHANDLER_CONNECTED);
194  while (!PLATFORM::CThread::IsStopped())
195  {
196  int r;
197  EventMessage msg;
198  r = m_event->RcvBackendMessage(EVENTHANDLER_TIMEOUT, msg);
199  if (r > 0)
200  DispatchEvent(msg);
201  else if (r < 0)
202  {
203  AnnounceStatus(EVENTHANDLER_DISCONNECTED);
204  RetryConnect();
205  }
206  else
207  {
208  AnnounceTimer();
209  }
210  }
211  AnnounceStatus(EVENTHANDLER_STOPPED);
212  // Close connection
213  m_event->Close();
214  return NULL;
215 }

236 void BasicEventHandler::RetryConnect()
237 {
238  unsigned c = 0;
239  while (!PLATFORM::CThread::IsStopped())
240  {
241  if (++c > 10) // 5 seconds
242  {
243  if (m_event->Open())
244  {
245  AnnounceStatus(EVENTHANDLER_CONNECTED);
246  break;
247  }
248  c = 0;
249  DBG(MYTH_DBG_INFO, "%s: could not open event socket (%d)\n", __FUNCTION__, m_event->GetSocketErrNo());
250  AnnounceStatus(EVENTHANDLER_NOTCONNECTED);
251  }
252  usleep(500000);
253  }
254 }
